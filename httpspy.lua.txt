function pick_request()
    local candidates = {
        syn and syn.request,
        http and http.request,
        rawget(_G, "http_request"),
        rawget(_G, "request"),
        getgenv().request
    }

    local selected
    local selectedScore = -1

    for _, candidate in ipairs(candidates) do
        if type(candidate) == "function" then
            local informationSuccess, information = pcall(getDebugInfo, candidate)
            local functionType = informationSuccess and information and information.what or "unknown"

            local hasUpvalue = false

            if getDebugUpvalue then
                local upvalueSuccess, firstUpvalue = pcall(getDebugUpvalue, candidate, 1)

                if upvalueSuccess and firstUpvalue ~= nil then
                    hasUpvalue = true
                end
            end

            local representation = tostring(candidate)
            local score = 0

            if functionType == "C" then
                score = score + 3
            end

            if not hasUpvalue then
                score = score + 2
            end

            if representation:find("function:") then
                score = score + 1
            end

            if score > selectedScore then
                selected = candidate
                selectedScore = score
            end
        end
    end

    return selected, selectedScore
end

local function copyTable(value)
    if type(value) ~= "table" then
        return value
    end

    local result = {}

    for key, child in pairs(value) do
        local copiedKey = type(key) == "string" and ("" .. key) or key

        if type(child) == "table" then
            result[copiedKey] = copyTable(child)
        elseif type(child) == "string" then
            result[copiedKey] = "" .. child
        else
            result[copiedKey] = child
        end
    end

    return result
end

function detect_httpspy()
    local findings = {}
    local requestFunction = pick_request()

    if not requestFunction then
        table.insert(findings, "no_request_ep")
        return true, findings
    end

    local informationSuccess, information = pcall(getDebugInfo, requestFunction)

    if informationSuccess and information then
        if information.what ~= "C" then
            table.insert(findings, "lua_wrapped_req")
        end

        if getDebugUpvalue then
            local upvalueSuccess, firstUpvalue = pcall(getDebugUpvalue, requestFunction, 1)

            if upvalueSuccess and firstUpvalue ~= nil then
                table.insert(findings, "req_has_upvalues")
            end
        end
    else
        table.insert(findings, "no_debuginfo")
    end

    local testArguments = {
        Url = "about:blank",
        Method = "GET",
        Headers = {
            A = "B"
        }
    }

    if getmetatable(testArguments) then
        table.insert(findings, "args_has_mt")
    end

    if not tostring(requestFunction):find("function:") then
        table.insert(findings, "req_tostring_mutated")
    end

    for _, ioFunction in ipairs({
        print,
        rconsoleprint,
        rconsolewarn,
        rconsoleerr,
        setclipboard
    }) do
        if type(ioFunction) == "function" then
            local success, ioInformation = pcall(getDebugInfo, ioFunction)

            if success and ioInformation and ioInformation.what == "Lua" then
                table.insert(findings, "io_wrapper")
                break
            end
        end
    end

    local namecallHooked = false

    if getrawmetatable then
        local metatable = getrawmetatable(game)

        if metatable and type(metatable.__namecall) == "function" then
            local success, namecallInformation = pcall(getDebugInfo, metatable.__namecall)

            if success and namecallInformation and namecallInformation.what == "Lua" then
                namecallHooked = true
            end
        end
    end

    if namecallHooked then
        table.insert(findings, "namecall_hook")
    end

    return #findings > 0, findings
end

local selectedRequest = pick_request()

local function protectedRequestCall(options)
    options = copyTable(options)

    if not selectedRequest then
        return {
            StatusCode = 0,
            Body = "no_request"
        }
    end

    local informationSuccess, information = pcall(getDebugInfo, selectedRequest)

    if informationSuccess and information and information.what ~= "C" then
        return {
            StatusCode = 0,
            Body = "wrapped_req"
        }
    end

    local success, result = pcall(selectedRequest, options)

    if not success then
        return {
            StatusCode = 0,
            Body = tostring(result)
        }
    end

    return result
end

local function checkRuntimeAnalysis()
    local environment = getgenv()
    local dumpedFunctionCount = 0

    for _, value in pairs(environment) do
        if type(value) == "function" then
            local success, result = pcall(function()
                local dump = string.dump(value)
                return dump:find("sanitize")
                    or dump:find("createlog")
                    or dump:find("notif")
            end)

            if success and result then
                dumpedFunctionCount = dumpedFunctionCount + 1
            end
        end
    end

    local wrappedBuiltinCount = 0

    if game.HttpService
        and game.HttpService.JSONDecode
        and not tostring(game.HttpService.JSONDecode):find("function:")
    then
        wrappedBuiltinCount = wrappedBuiltinCount + 1
    end

    if identifyexecutor
        and not tostring(identifyexecutor):find("function:")
    then
        wrappedBuiltinCount = wrappedBuiltinCount + 1
    end

    local analytics = game:GetService("RbxAnalyticsService")

    if analytics and analytics.GetClientId then
        local success, clientId = pcall(function()
            return analytics:GetClientId()
        end)

        if success and type(clientId) == "string" then
            for _, value in pairs(environment) do
                if type(value) == "string" and value:find(clientId) then
                    wrappedBuiltinCount = wrappedBuiltinCount + 2
                end
            end
        end
    end

    return dumpedFunctionCount >= 2 or wrappedBuiltinCount >= 2
end

local function checkEnvironmentContamination()

    local score = 0
    local expectedKeys = {
        "uilib",
        "executor",
        "version",
        "functions",
        "band",
        "bxor",
        "bnot",
        "lshift",
        "rshift",
        "rshift1",
        "random",
        "sub",
        "floor",
        "fakehwid",
        "fakeip",
        "actualhwid",
        "actualip",
        "sanitize",
        "search",
        "serializer",
        "hook"
    }

    for _, key in pairs(expectedKeys) do
        if rawget(environment, key) then
            score = score + 1
        end
    end

    if rawget(environment, "hookfunction")
        and rawget(environment, "newcclosure")
    then
        score = score + 2
    end

    if rawget(environment, "identifyexecutor") then
        score = score + 1
    end

    for _, value in pairs(environment) do
        if type(value) == "table" then
            local matchingNames = 0

            for key in pairs(value) do
                if type(key) == "string" then
                    local lowered = key:lower()

                    if lowered:find("http")
                        or lowered:find("request")
                        or lowered:find("spy")
                        or lowered:find("log")
                        or lowered:find("sanitize")
                    then
                        matchingNames = matchingNames + 1
                    end
                end
            end

            if matchingNames >= 3 then
                score = score + 2
            end
        end
    end

    return score >= 5
end

local function installRuntimeMonitors()
    local scanInterval = 0.5
    local lastScan = 0
    local consecutiveDetections = 0

    RunService.Heartbeat:Connect(function()
        if tick() - lastScan >= scanInterval then
            lastScan = tick()

            local detected, findings = scanEnvironment()

            if detected then
                consecutiveDetections = consecutiveDetections + 1

                if consecutiveDetections >= 2 then
                    terminateClient(
                        "E-Trace ANTI-HOOK: i'm sorry but you can't hook me bro <3 ["
                            .. table.concat(findings, "|")
                            .. "]"
                    )
                end
            else
                consecutiveDetections = 0
            end
        end
    end)

    if getgenv and getgenv().loadstring then
        local originalEnvironmentLoadstring = getgenv().loadstring

        getgenv().loadstring = function(source, ...)
            if type(source) == "string" then
                local lowered = source:lower()

                if lowered:find("knockzoo")
                    or lowered:find("http%-spy")
                    or lowered:find("createlog")
                    or lowered:find("uilib")
                then
                    terminateClient("E-Trace ANTI-HOOK: i'm sorry but you can't hook me bro <3")
                end
            end

            return originalEnvironmentLoadstring(source, ...)
        end
    end

    local knownEnvironmentKeys = {}

    for key in pairs(getgenv()) do
        knownEnvironmentKeys[key] = true
    end

    spawn(function()
        while wait(2) do
            for key in pairs(getgenv()) do
                if not knownEnvironmentKeys[key] then
                    if type(key) == "string" then
                        local lowered = key:lower()

                        if lowered:find("uilib")
                            or lowered:find("serializer")
                            or lowered:find("httpspy")
                        then
                            terminateClient(
                                "E-Trace ANTI-HOOK: i'm sorry but you can't hook me bro <3 ["
                                    .. key
                                    .. "]"
                            )
                        end
                    end

                    knownEnvironmentKeys[key] = true
                end
            end
        end
    end)
end

if loadstring then

local function periodicRunner()
    while true do
        local success, runnerError = pcall(function()
            if type(immediateDetection) == "function" then
                local detectionSuccess, detectionResult = pcall(immediateDetection)

                if detectionSuccess and detectionResult then
                    pcall(function()
                        if type(detectionResult) == "table" then
                            Player:Kick("E-Trace: immediate [" .. table.concat(detectionResult, "|") .. "]")
                        else
                            Player:Kick("E-Trace: immediate [" .. tostring(detectionResult) .. "]")
                        end
                    end)
                end
            end

            if type(detect_httpspy) == "function" then
                local detectionSuccess, detectionResult = pcall(detect_httpspy)

                if detectionSuccess and detectionResult then
                    pcall(function()
                        if type(detectionResult) == "table" then
                            Player:Kick("E-Trace: httpspy [" .. table.concat(detectionResult, "|") .. "]")
                        else
                            Player:Kick("E-Trace: httpspy")
                        end
                    end)
                end
            end

            if type(detectHTTPSpy) == "function" then
                local detectionSuccess, detectionResult = pcall(detectHTTPSpy)

                if detectionSuccess and detectionResult then
                    pcall(function()
                        if type(detectionResult) == "table" then
                            Player:Kick("E-Trace: httpSpy [" .. table.concat(detectionResult, "|") .. "]")
                        else
                            Player:Kick("E-Trace: httpSpy")
                        end
                    end)
                end
            end

            if type(detectEDAntiKick) == "function" then
                local detectionSuccess, detectionResult = pcall(detectEDAntiKick)

                if detectionSuccess and detectionResult then
                    pcall(function()
                        Player:Kick("E-Trace: anti-kick detected")
                    end)
                end
            end

            if type(pick_request) == "function" then
                local requestSuccess, requestFunction, requestScore = pcall(pick_request)

                if requestSuccess
                    and (not requestFunction
                        or (type(requestScore) == "number" and requestScore < 1))
                then
                    pcall(function()
                        Player:Kick("E-Trace: request entry suspicious")
                    end)
                end
            end

            local environmentSuccess, environmentFlagged = pcall(function()
                local environment = getgenv()
                return rawget(environment, "uilib")
                    or rawget(environment, "serializer")
                    or rawget(environment, "WebStealer")
            end)

            if environmentSuccess and environmentFlagged then
                pcall(function()
                    Player:Kick("E-Trace: env flagged")
                end)
            end
        end)

        if not success then
            warn("E-Trace periodic runner error:", runnerError)
        end

        task.wait(3)
    end
end

task.spawn(periodicRunner)

wait(1)

local detected, findings = scanEnvironment()

if detected then
    terminateClient(
        "E-Trace ANTI-HOOK: i'm sorry but you can't hook me bro <3 ["
            .. table.concat(findings, "|")
            .. "]"
    )
else
    installRuntimeMonitors()
end
